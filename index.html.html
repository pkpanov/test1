Infinite Rows SplitFlapBoard Viewport Demo

This is an extended version of the previous demo. The canvas is only a viewport that shows 3.5 rows, while the virtual board has infinitely repeating rows. Drag inside the canvas to scroll through the virtual rows. Start runs a 3-second timer that animates only the currently visible rows.

Live demo

Drag vertically inside the canvas to scroll. The canvas height is rowHeight * 3.5, so it shows three full rows and half of the next row.

What changed from the previous one-row demo?

Before

There was one row with three tiles. The button animated those three tile objects directly.

Now

There are virtual infinite rows. The code computes which row indices are visible, creates only those row tile objects, renders only them, and animates only them.

Important: the canvas is not actually infinite. The canvas is a fixed-size viewport. The virtual world is infinite because scrollY can keep increasing or decreasing, and visible row indices are calculated from scrollY.

Connection diagram

The core idea

The board does not store infinite rows. It stores only a scroll position and a small row cache.

const first = Math.floor(scrollY / rowHeight);

// If viewport shows 3.5 rows, this usually gives 4 row indices.

Source code — AnimationLoop

AnimationLoop.js
class AnimationLoop {
  constructor(onFrame) {
    // onFrame is a callback given by SplitFlapBoard.
    // In this demo it will be: (dt, now) => this.frame(dt, now)
    this.onFrame = onFrame;

    // true while requestAnimationFrame should continue looping.
    this.running = false;

    // id returned by requestAnimationFrame; needed for cancelAnimationFrame.
    this.frameId = null;

    // Time of previous frame, used to calculate dt.
    this.lastTime = 0;
  }

  start() {
    // Avoid accidentally starting two loops.
    if (this.running) return;

    this.running = true;
    this.lastTime = performance.now();

    // Arrow function keeps this AnimationLoop object as "this".
    const frame = (now) => {
      if (!this.running) return;

      // Convert milliseconds to seconds.
      // Cap dt to avoid huge jumps when tab was inactive.
      const dt = Math.min((now - this.lastTime) / 1000, 0.05);
      this.lastTime = now;

      // This is the important connection:
      // AnimationLoop -> SplitFlapBoard.frame(dt, now)
      this.onFrame(dt, now);

      // Ask browser for the next visual frame.
      this.frameId = requestAnimationFrame(frame);
    };

    this.frameId = requestAnimationFrame(frame);
  }

  stop() {
    this.running = false;

    if (this.frameId !== null) {
      cancelAnimationFrame(this.frameId);
      this.frameId = null;
    }
  }
}

Source code — CanvasRenderer

CanvasRenderer.js
class CanvasRenderer {
  constructor(canvas, options = {}) {
    this.canvas = canvas;
    this.ctx = canvas.getContext("2d");

    this.options = {
      width: 640,
      height: 455,
      background: "#10131a",
      rowHeight: 130,
      ...options
    };

    this.resize(this.options.width, this.options.height);
  }

  resize(width, height) {
    const dpr = window.devicePixelRatio || 1;

    this.width = width;
    this.height = height;

    // CSS size: how large the canvas appears.
    this.canvas.style.width = width + "px";
    this.canvas.style.height = height + "px";

    // Real bitmap size: how many physical pixels canvas has.
    this.canvas.width = Math.round(width * dpr);
    this.canvas.height = Math.round(height * dpr);

    // Allows drawing code to still use CSS-pixel coordinates.
    this.ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  }

  render(view) {
    /*
      view is not the infinite world.
      view contains only the currently visible rows.

      Example shape:

      {
        width: 640,
        height: 455,
        scrollY: 260,
        rowHeight: 130,
        rows: [
          { index: 2, y: 0, tiles: [...] },
          { index: 3, y: 130, tiles: [...] },
          ...
        ]
      }
    */

    const ctx = this.ctx;

    // Clear old frame.
    ctx.clearRect(0, 0, this.width, this.height);

    // Background.
    ctx.fillStyle = this.options.background;
    ctx.fillRect(0, 0, this.width, this.height);

    // Draw only rows that are inside the viewport.
    for (const row of view.rows) {
      this.drawRow(row, view);
    }

    this.drawViewportInfo(view);
  }

  drawRow(row, view) {
    const ctx = this.ctx;
    const rowY = row.y;

    ctx.save();

    // Row background, useful to see the repeated virtual rows.
    ctx.fillStyle = row.index % 2 === 0
      ? "rgba(255,255,255,0.035)"
      : "rgba(255,255,255,0.015)";
    ctx.fillRect(0, rowY, this.width, view.rowHeight);

    // Row separator line.
    ctx.strokeStyle = "rgba(255,255,255,0.08)";
    ctx.beginPath();
    ctx.moveTo(0, rowY);
    ctx.lineTo(this.width, rowY);
    ctx.stroke();

    // Debug row index label.
    ctx.fillStyle = "rgba(255,255,255,0.35)";
    ctx.font = "13px system-ui";
    ctx.textAlign = "left";
    ctx.textBaseline = "top";
    ctx.fillText("row " + row.index, 10, rowY + 8);

    // Draw the three tiles in this row.
    for (const tile of row.tiles) {
      this.drawTile(tile, rowY);
    }

    ctx.restore();
  }

  drawTile(tile, rowY) {
    const ctx = this.ctx;

    // tile.y is local inside the row.
    // rowY is where this virtual row appears in the viewport.
    const x = tile.x;
    const y = rowY + tile.y;

    ctx.save();

    // Tile body.
    ctx.fillStyle = "#27324a";
    ctx.fillRect(x, y, tile.width, tile.height);

    // Top half highlight.
    ctx.fillStyle = "rgba(255,255,255,0.06)";
    ctx.fillRect(x, y, tile.width, tile.height / 2);

    // Split line.
    ctx.fillStyle = "#05070b";
    ctx.fillRect(x, y + tile.height / 2 - 1, tile.width, 2);

    // Border.
    ctx.strokeStyle = "#697894";
    ctx.lineWidth = 2;
    ctx.strokeRect(x, y, tile.width, tile.height);

    // Letter.
    ctx.fillStyle = "#ffffff";
    ctx.font = "bold 48px system-ui";
    ctx.textAlign = "center";
    ctx.textBaseline = "middle";
    ctx.fillText(tile.label, x + tile.width / 2, y + tile.height / 2 + 1);

    ctx.restore();
  }

  drawViewportInfo(view) {
    const ctx = this.ctx;

    ctx.save();

    // Small overlay so we can see what the virtual viewport is doing.
    ctx.fillStyle = "rgba(0,0,0,0.45)";
    ctx.fillRect(12, this.height - 40, 390, 28);

    ctx.fillStyle = "rgba(255,255,255,0.9)";
    ctx.font = "13px system-ui";
    ctx.textAlign = "left";
    ctx.textBaseline = "middle";

    const first = view.rows[0]?.index;
    const last = view.rows[view.rows.length - 1]?.index;

    ctx.fillText(
      `scrollY=${view.scrollY.toFixed(1)} | visible rows: ${first} ... ${last}`,
      24,
      this.height - 26
    );

    ctx.restore();
  }
}

Source code — SplitFlapBoard

SplitFlapBoard.js
class SplitFlapBoard {
  constructor(canvas, options = {}) {
    this.options = {
      width: 640,

      // One row is 130px high.
      rowHeight: 130,

      // 3.5 rows visible:
      // 130 * 3.5 = 455
      height: 130 * 3.5,

      background: "#10131a",
      ...options
    };

    /*
      We DO NOT create infinite rows.

      Instead, we store only:
      - scrollY: where the viewport is in the virtual world
      - rowCache: tile objects for rows that are currently visible

      This is the key "infinite canvas" idea.
    */
    this.scrollY = 0;
    this.rowCache = new Map();

    /*
      These are the three base tiles repeated in every virtual row.

      Each row gets A, B, C.
      Only x changes during animation.
      y is local inside a row.
    */
    this.tileDefinitions = [
      { id: "tileA", label: "A", baseX: 50,  y: 18, width: 80, height: 100 },
      { id: "tileB", label: "B", baseX: 160, y: 18, width: 80, height: 100 },
      { id: "tileC", label: "C", baseX: 270, y: 18, width: 80, height: 100 }
    ];

    // Active x-position animations for visible-row tiles.
    this.animations = [];

    // dirty means "redraw on next frame".
    this.dirty = true;

    // Timer used by the Start/Stop buttons.
    // This is separate from requestAnimationFrame.
    this.autoTimerId = null;

    // Drag-scroll state.
    this.isDragging = false;
    this.lastPointerY = 0;

    // SplitFlapBoard owns renderer and animation loop.
    this.renderer = new CanvasRenderer(canvas, this.options);

    this.loop = new AnimationLoop((dt, now) => {
      this.frame(dt, now);
    });

    this.installDragScrolling(canvas);
  }

  start() {
    // Start the visual animation loop.
    // This loop is needed for rendering and smooth animations.
    this.loop.start();
    this.requestRender();
  }

  stop() {
    // Stop auto timer and animation loop.
    this.stopAutoAnimation();
    this.loop.stop();
  }

  startAutoAnimation() {
    /*
      This is what the Start button calls.

      Every 3 seconds, animate only the currently visible rows.
    */

    // Ensure visual loop is running.
    this.start();

    // Do not create two timers.
    if (this.autoTimerId !== null) return;

    // Do one animation immediately.
    this.animateVisibleTiles();

    // Then repeat every 3000 ms = 3 seconds.
    this.autoTimerId = setInterval(() => {
      this.animateVisibleTiles();
    }, 3000);
  }

  stopAutoAnimation() {
    if (this.autoTimerId !== null) {
      clearInterval(this.autoTimerId);
      this.autoTimerId = null;
    }
  }

  requestRender() {
    // This does not draw immediately.
    // It only marks that next frame should render.
    this.dirty = true;
  }

  frame(dt, now) {
    // AnimationLoop calls this method every browser frame.
    this.updateAnimations(dt);

    // Draw only when necessary.
    if (this.dirty || this.animations.length > 0) {
      this.render();
      this.dirty = false;
    }
  }

  render() {
    /*
      Build a small "view model" that contains only visible rows.

      CanvasRenderer receives this small object.
      It never receives infinite data.
    */
    const view = this.createVisibleView();

    this.renderer.render(view);
  }

  createVisibleView() {
    const visibleIndices = this.getVisibleRowIndices();

    // Remove cached rows that are no longer visible.
    // This keeps memory small and proves we are not updating infinity.
    this.pruneInvisibleRows(visibleIndices);

    const rows = visibleIndices.map((rowIndex) => {
      return {
        index: rowIndex,

        // Convert virtual row position to viewport position.
        // Example:
        // rowIndex * rowHeight = row position in infinite world
        // subtract scrollY to place it on screen.
        y: rowIndex * this.options.rowHeight - this.scrollY,

        tiles: this.getRowTiles(rowIndex)
      };
    });

    return {
      width: this.options.width,
      height: this.options.height,
      rowHeight: this.options.rowHeight,
      scrollY: this.scrollY,
      rows
    };
  }

  getVisibleRowIndices() {
    const rowHeight = this.options.rowHeight;
    const top = this.scrollY;
    const bottom = this.scrollY + this.options.height;

    // first row that touches the viewport.
    const first = Math.floor(top / rowHeight);

    // last row that touches the viewport.
    const last = Math.floor(bottom / rowHeight);

    const indices = [];

    for (let rowIndex = first; rowIndex <= last; rowIndex++) {
      indices.push(rowIndex);
    }

    return indices;
  }

  getRowTiles(rowIndex) {
    /*
      rowCache stores only rows that have appeared in the viewport.

      Key: row index number.
      Value: array of the three tile objects for that row.
    */
    if (!this.rowCache.has(rowIndex)) {
      const tiles = this.tileDefinitions.map((definition) => {
        return {
          rowIndex,
          id: definition.id,
          label: definition.label,

          // x starts at baseX, but can animate.
          x: definition.baseX,
          baseX: definition.baseX,

          y: definition.y,
          width: definition.width,
          height: definition.height
        };
      });

      this.rowCache.set(rowIndex, tiles);
    }

    return this.rowCache.get(rowIndex);
  }

  pruneInvisibleRows(visibleIndices) {
    const visibleSet = new Set(visibleIndices);

    for (const rowIndex of this.rowCache.keys()) {
      if (!visibleSet.has(rowIndex)) {
        this.rowCache.delete(rowIndex);
      }
    }

    // Also remove animations for rows that are no longer visible.
    this.animations = this.animations.filter((animation) => {
      return visibleSet.has(animation.tile.rowIndex);
    });
  }

  animateVisibleTiles() {
    /*
      This is called by the timer every 3 seconds.

      It deliberately uses getVisibleRowIndices().
      That means it touches only the rows currently in the viewport.
    */
    const visibleRows = this.getVisibleRowIndices();

    for (const rowIndex of visibleRows) {
      const tiles = this.getRowTiles(rowIndex);

      for (const tile of tiles) {
        // Move around the base position, but keep tile inside canvas.
        const randomOffset = Math.random() * 130 - 25;
        const targetX = this.clamp(
          tile.baseX + randomOffset,
          20,
          this.options.width - tile.width - 20
        );

        this.animateTileX(tile, targetX, 0.7 + Math.random() * 0.45);
      }
    }
  }

  animateTileX(tile, targetX, duration = 0.8) {
    /*
      This version receives the actual visible-row tile object.

      That is important:
      we are not searching through infinite rows.
      We already have a visible tile object.
    */

    // Remove any previous x animation for this same visible tile.
    this.animations = this.animations.filter((animation) => {
      return animation.tile !== tile;
    });

    this.animations.push({
      tile,
      fromX: tile.x,
      toX: targetX,
      elapsed: 0,
      duration
    });

    this.requestRender();
  }

  updateAnimations(dt) {
    if (this.animations.length === 0) return;

    const stillRunning = [];

    for (const animation of this.animations) {
      animation.elapsed += dt;

      const progress = Math.min(animation.elapsed / animation.duration, 1);
      const eased = this.easeOutCubic(progress);

      animation.tile.x = animation.fromX +
        (animation.toX - animation.fromX) * eased;

      if (progress < 1) {
        stillRunning.push(animation);
      }
    }

    this.animations = stillRunning;
    this.requestRender();
  }

  installDragScrolling(canvas) {
    // touch-action none prevents mobile browser scrolling/zooming
    // while dragging on canvas.
    canvas.style.touchAction = "none";

    canvas.addEventListener("pointerdown", (event) => {
      this.isDragging = true;
      this.lastPointerY = event.clientY;
      canvas.setPointerCapture?.(event.pointerId);
      canvas.classList.add("dragging");
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!this.isDragging) return;

      const dy = event.clientY - this.lastPointerY;
      this.lastPointerY = event.clientY;

      /*
        Dragging the pointer down moves the rows down.
        To make the content move with the finger/mouse,
        we subtract dy from scrollY.

        scrollY is allowed to become negative.
        That means the virtual canvas is infinite in both directions.
      */
      this.scrollY -= dy;

      this.requestRender();
    });

    const endDrag = (event) => {
      this.isDragging = false;
      canvas.releasePointerCapture?.(event.pointerId);
      canvas.classList.remove("dragging");
    };

    canvas.addEventListener("pointerup", endDrag);
    canvas.addEventListener("pointercancel", endDrag);
  }

  easeOutCubic(t) {
    return 1 - Math.pow(1 - t, 3);
  }

  clamp(value, min, max) {
    return Math.max(min, Math.min(max, value));
  }
}

const canvas = document.querySelector("#board");
const startButton = document.querySelector("#startButton");
const stopButton = document.querySelector("#stopButton");

const board = new SplitFlapBoard(canvas, {
  width: 640,
  rowHeight: 130,
  height: 130 * 3.5,
  background: "#10131a"
});

// Start the visual loop so dragging can render even before the timer starts.

  board.startAutoAnimation();
});

  board.stopAutoAnimation();
});

What “only visible rows are updated” means here

The timer calls animateVisibleTiles(). That method first calls getVisibleRowIndices(). It never loops through all rows, because there is no array containing all rows.

The important method
animateVisibleTiles() {
  const visibleRows = this.getVisibleRowIndices();

  for (const rowIndex of visibleRows) {
    const tiles = this.getRowTiles(rowIndex);

    for (const tile of tiles) {
      this.animateTileX(tile, targetX, duration);
    }
  }
}
Note: rows that leave the viewport are pruned from rowCache, and their animations are removed. This keeps the demo close to the idea of an infinite virtual canvas with a small real viewport.
